169def creerChoixCouleur(iCoordonneeX: int, iCoordonneeY: int, sNomLabel: str, iJoueur: int) -> None:
170 """
171 @brief Créé le widget permettants de choisir une couleur et de l'afficher
172 @param iCoordonneeX coordonnée en abscisse où doit être placer le widget
173 @param iCoordonneeY coordonnée en ordonnée où doit être placer le widget
174 @param sNomLabel texte sur le label devant le widget de choix de couleur
175 @param iJoueur joueur qui a demander un changement de couleur (1 pour J1, 2 pour J2)
176 """
177
178 global toplevelFenetreParametre
179 global sCouleurJetonJ1
180 global sCouleurJetonJ2
181
182 frameCouleur: tk.Frame = tk.Frame(toplevelFenetreParametre)
183 frameCouleur.configure(height=50, width=350)
184 frameCouleur.place(anchor="nw", rely=iCoordonneeX, relx=iCoordonneeY)
185
186 # Créé le label qui indique le joueur lié à la couleur
187 labelCouleur: tk.Label = tk.Label(frameCouleur)
188 labelCouleur.configure(font="{Arial} 16 {underline}", text=sNomLabel)
189 labelCouleur.place(anchor="nw", relx=0.08, rely=0.27, x=0, y=0)
190
191 canvaCouleur: tk.Canvas = tk.Canvas(frameCouleur)
192 canvaCouleur.configure(cursor="pencil", height=75, width=75)
193
194 # Affiche la couleur du jeton du joueur qui a cliqué sur le bouton
195 if (iJoueur == 1):
196 canvaCouleur.create_oval(33, 33, 5, 5, outline="black", fill=sCouleurJetonJ1)
197 else:
198 canvaCouleur.create_oval(33, 33, 5, 5, outline="black", fill=sCouleurJetonJ2)
199
200 # Lie la fonction qui permet de choisir la couleur du jeton avec la couleur affichée
201 canvaCouleur.bind("<Button-1>", lambda event, canvaCouleur=canvaCouleur, iJoueur=iJoueur: choisirCouleur(canvaCouleur, iJoueur))
202 canvaCouleur.place(relx=0.9, rely=0.13, x=0, y=0)
